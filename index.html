function getParticipants(handshakes) {
    let amountOfParticipants = 1;
    let indexParticipants = 1;
                    
    while (handshakes) {
        if (amountOfParticipants === indexParticipants) {
            amountOfParticipants++;
            indexParticipants = 1;
        }
        else {
        handshakes--;
        indexParticipants++;
        }
    }
    
    return amountOfParticipants;
}
